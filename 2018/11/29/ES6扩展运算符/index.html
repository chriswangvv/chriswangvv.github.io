ES6扩展运算符...

发表于 2018-11-29 | 分类于 ES

我第一次见这个扩展运算符是在比较大小的一道题里(highAndLow("4 5 29 54 4 0 -214 542 -64 1 -3 6 -6"))

function highAndLow(numbers){
  var arr = numbers.split(" ").map(Number);
  console.log(Math.max(...arr));
  return Math.max(...arr)
}

可用的地方

可变参数个数的函数调用

window.onload = function() {
        let arr = [1, 2, 3, 4, 5, 6]
        let sum = add(...arr)
        let items = [4, 9, 8]
        let result = push(arr, ...items)
        console.log(result); // [1, 2, 3, 4, 5, 6, 4, 9, 8, 4, 9, 8]
        console.log(sum); // 21
}

function push(arr, ...items) {
    array.push(...items); // ...items === 4,9,8
    return array
}

function add(...vals) {
    let sum = 0;
    for (let i = 0; i < vals.lenght; i++) {
        sum += vals[i];
    }
    return sum;
}

数组合并*

let arr1 = [1, 2, 3]
let arr2 = [9, 8, 7]
let newArr = [20]
// 旧写法：
newArr = newArr.concat(arr1).concat(arr2)
// 新写法：
newArr = [20, ...arr1, ...arr2]

代替es5的apply方法

  • apply

    function f(x, y, z) {
    // ...
}
var args = [0, 1, 2]

// es5：
f.apply(null, args)

// es6:
f(...args)

  • Math.max/min

    let arr1 = [0, 1, 2, -51, 98]

// es5:
Math.max.apply(null, arr1)
// es6:
console.log(Math.min(...arr1))

  • push到尾部

    let arr1 = [0, 1, 2];
let arr2 = [9, 8, 7];

// es5:
Array.prototype.push.apply(arr1, arr2);  
// es6
arr2.push(...arr1)

  • 新建Date类型

    // ES5  
new (Date.bind.apply(Date, [null, 2015, 1, 1]))  
// ES6  
new Date(...[2015, 1, 1]);

与解构赋值结合，生成新数组

// ES5  
a = list[0], rest = list.slice(1)  
// ES6  
[a, ...rest] = list 
下面是另外一些例子。  
const [first, ...rest] = [1, 2, 3, 4, 5];  
first // 1  
rest // [2, 3, 4, 5]  
const [first, ...rest] = [];  
first // undefined  
rest // []:  
const [first, ...rest] = ["foo"];  
first // "foo"  
rest // []

将字符串转为真正的数组

[...'hello']  // [ "h", "e", "l", "l", "o" ]

备注： 数组转为字符串：

arr.join[""] // 引号内为自定义的分隔符

将实现了 Iterator 接口的对象转为数组

var nodeList = document.querySelectorAll('div');  
var array = [...nodeList];

备注

数组合并的两个常用方法

  • concat: arr1.concat(arr2).concat(arr3)

    1. arr1的结果顺序是arr1+arr2+arr3

    2. 可以用这个复制一份数组: arr1.concat(arr1)

    3. 数组长度无限制，返回值是合并后的arr1

  • Array.prototype.push.apply(arr1,arr2) / arr1.push.apply(arr1,arr2)

    1. arr1和arr2都变成合并后的数组

    2. 数组有长度限制，一般不超过十万

    3. 返回值是合并后数组的长度

    4. 在一万数据左右性能和concat差不多，大约2毫秒
