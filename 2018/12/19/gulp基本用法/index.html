gulp基本用法

发表于 2018-12-19 | 分类于 工具

node版本： v10.14.2
gulp版本： CLI version 2.0.1

压缩JS

  1. 安装gulp、 gulp-uglify

  2. 编写代码

    var gulp = require('gulp'); // 获取模块
var uglify = require('gulp-uglify'); // 用于压缩js

gulp.task('script', function() {
    gulp.src('js/*.js').pipe(uglify()).pipe(gulp.dest('dist/js'));
    return new Promise(function(resolve, reject) {
        console.log("HTTP Server Started");
        resolve();
    });
});
gulp.task('auto', function() { // watch 监听文件修改
    gulp.watch('js/*.js', gulp.series('script'))
});
gulp.task('default', gulp.series('script', 'auto')) // 默认任务

  3. 命令行执行任务

压缩CSS

目前很多教程里（包括近期的很多教程）用的都是gulp-minify-css`，但是这个包已经被弃用了

  1. 安装

  2. 编写代码

    var gulp = require('gulp'); // 获取模块
var cleanCss = require('gulp-clean-css');

gulp.task('css', function() {
    gulp.src('css/*.css').pipe(cleanCss()).pipe(gulp.dest('dist/css'));
    return new Promise(function(resolve, reject) {
        console.log('CSS HTTP Server Started');
        resolve();
    })
});
gulp.task('auto', function() {
    gulp.watch('css/*.css', gulp.series(['css']));
});
gulp.task('default', gulp.series(['auto', 'css']));

  3. 命令行执行任务

压缩图片

  1. 安装gulp-imagemin

    缩小PNG，JPEG，GIF和SVG图像
    （后缀名是jpg的不能被压缩，所以还是得注意一下文件格式）

  2. 编写代码

    gulp.src('img/*.*').pipe(imagemin({ progressive: true })).pipe(gulp.dest('dist/img'));

  3. 命令行执行任务
